362class IAMPolicyStatement(object): 

363 def __init__(self, statement): 

364 self._statement = statement

366 def is_action_permitted(self, action): 

367 is_action_concerned = False 

368 

369 if "NotAction" in self._statement: 

370 if not self._check_element_matches("NotAction", action): 

371 is_action_concerned = True 

372 else: # Action is present 

373 if self._check_element_matches("Action", action): 

374 is_action_concerned = True 

375 

376 if is_action_concerned: 

377 if self._statement["Effect"] == "Allow": 

378 return PermissionResult.PERMITTED 

379 else: # Deny 

380 return PermissionResult.DENIED 

381 else: 

382 return PermissionResult.NEUTRAL

384 def _check_element_matches(self, statement_element, value): 

385 if isinstance(self._statement[statement_element], list): 

386 for statement_element_value in self._statement[statement_element]: 

387 if self._match(statement_element_value, value): 

388 return True 

389 return False 

390 else: # string 

391 return self._match(self._statement[statement_element], value)
